def generate(options={})
  options ||= {}

  dir = nil # defaults to curent directory

  options[:trial]  = trial?
  options[:debug] = debug?
  options[:quiet] = quiet?
  options[:force] = force?

  tiller = Erbside::Runner.new(dir, options)
  tiller.till
end
